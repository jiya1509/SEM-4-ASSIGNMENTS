DECLARE
    num      NUMBER;    
    digit    NUMBER;      
    sum      NUMBER := 0; 
BEGIN
    num := &Enter_Number;
    WHILE num != 0 LOOP
        digit := MOD(num, 10); 
        sum := sum + digit;   
        num := TRUNC(num / 10);
    END LOOP;

    DBMS_OUTPUT.PUT_LINE('The sum of the digits is: ' || sum);
END;
/
