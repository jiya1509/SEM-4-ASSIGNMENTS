DECLARE
    num      NUMBER := &num; 
    reverse  NUMBER := 0;    
    digit    NUMBER;         
BEGIN
    WHILE num != 0 LOOP
        digit := MOD(num, 10);        
        reverse := reverse * 10 + digit;
        num := TRUNC(num / 10);       
    END LOOP;

    DBMS_OUTPUT.PUT_LINE('The reverse of the number is: ' || reverse);
END;
/
